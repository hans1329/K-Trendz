// Simplified burn mechanism
function burnForStars(uint256 tokenAmount) external {
    require(tokenAmount > 0, "Amount must be positive");
    
    // Calculate Stars based on dynamic rate
    uint256 exchangeRate = calculateDynamicRate();
    uint256 starsReceived = tokenAmount * exchangeRate;
    
    // Burn tokens permanently
    _burn(msg.sender, tokenAmount);
    
    // Credit Stars to user profile (off-chain via backend)
    emit TokensBurned(msg.sender, tokenAmount, starsReceived, exchangeRate);
    
    // Backend listener adds Stars to user's platform account
}

function calculateDynamicRate() public view returns (uint256) {
    uint256 totalBurned = initialSupply - totalSupply(); // Track burned amount
    uint256 maxSupply = 5_000_000_000 * 10**18;
    
    // Base rate: 10 Stars per KTREND
    uint256 baseRate = 10;
    
    // Premium increases as more tokens are burned (up to 50% bonus)
    uint256 premiumFactor = (totalBurned * 50) / (maxSupply * 100);
    
    return baseRate * (100 + premiumFactor) / 100;
}
